% Part formatting for separate research sections
% Make parts look like separate documents with clear visual separation
\makeatletter
\renewcommand\part{%
  \if@openright\cleardoublepage\else\clearpage\fi
  \thispagestyle{empty}%
  \if@twocolumn
    \onecolumn
    \@tempswatrue
  \else
    \@tempswafalse
  \fi
  \null\vfil
  \secdef\@part\@spart}
\def\@part[#1]#2{%
    \ifnum \c@secnumdepth >-2\relax
      \refstepcounter{part}%
      \addcontentsline{toc}{part}{\thepart\hspace{1em}#1}%
    \else
      \addcontentsline{toc}{part}{#1}%
    \fi
    \markboth{}{}%
    \vspace*{2cm}
    {\centering
     \interlinepenalty \@M
     \normalfont
     \ifnum \c@secnumdepth >-2\relax
       \fontsize{24}{28}\selectfont\bfseries \partname\nobreakspace\thepart
       \par
       \vskip 30\p@
     \fi
     \fontsize{28}{34}\selectfont \bfseries #2\par}%
    \vspace{3cm}
    \@endpart}
\def\@spart#1{%
    \vspace*{2cm}
    {\centering
     \interlinepenalty \@M
     \normalfont
     \fontsize{28}{34}\selectfont \bfseries #1\par}%
    \vspace{3cm}
    \@endpart}
\def\@endpart{\vfil\newpage
              \if@twocolumn
                \if@tempswa
                  \twocolumn
                \fi
              \fi
              \thispagestyle{plain}}
\makeatother
